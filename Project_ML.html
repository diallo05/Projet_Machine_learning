import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from scipy.stats import chi2_contingency

# Fonction pour calculer le Cramer's V
def cramers_v(confusion_matrix):
    chi2, p, dof, expected = chi2_contingency(confusion_matrix)
    n = confusion_matrix.sum().sum()
    r, k = confusion_matrix.shape
    return np.sqrt(chi2 / (n * (min(k - 1, r - 1))))

# Fonction pour crer une matrice de Cramer's V pour les colonnes spcifies
def cramer_v_matrix(df_quanti, cols):
    n = len(cols)
    matrix = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i, n):
            if i == j:
                matrix[i, j] = 1.0 
            else:
                contingency_table = pd.crosstab(df_quanti[cols[i]], df_quanti[cols[j]])
                matrix[i, j] = cramers_v(contingency_table)
                matrix[j, i] = matrix[i, j]
    
    return pd.DataFrame(matrix, index=cols, columns=cols)

# Liste des colonnes  analyser
cols = df_quanti.columns

# Calcul de la matrice de Cramer's V
cramers_v_df = cramer_v_matrix(df_quanti, cols)

# Visualisation de la matrice de Cramer's V avec une heatmap
plt.figure(figsize=(10, 8))
sns.heatmap(cramers_v_df, annot=True, cmap='Blues', linewidths=0.5, vmin=0, vmax=1)
plt.title("Matrice de Dpendance - Cramer's V", fontsize=16)

def predict_poverty(wilaya, taille, typmen, sitmatcm, nivcm, aliment, depeduc, autdep, depsante, logement, transp):
    try:
        # Convertir les entres utilisateur en un DataFrame
        input_data = pd.DataFrame({
            "wilaya": [wilaya],
            "taille": [taille],
            "typmen": [typmen],
            "sitmatcm": [sitmatcm],
            "nivcm": [nivcm],
            "aliment": [aliment],
            "depeduc": [depeduc],
            "autdep": [autdep],
            "depsante": [depsante],
            "logement": [logement],
            "transp": [transp],
        })

        # Encodage des variables catgoriques
        input_data = pd.get_dummies(input_data)
        for col in X.columns:
            if col not in input_data.columns:
                input_data[col] = 0  # Ajouter les colonnes manquantes

        # Rorganiser les colonnes
        input_data = input_data[X.columns]

        # Normalisation
        input_data_scaled = scaler.transform(input_data)

        # Prdiction
        prediction = model.predict(input_data_scaled)
        prob = model.predict_proba(input_data_scaled)[0][1]

        return "Pauvre" if prediction[0] == 1 else "Non Pauvre", f"Probabilit : {prob:.2f}"
    except Exception as e:
        return "Erreur", str(e)
